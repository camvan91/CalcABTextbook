\begin{center}
	{\Large Exercises}
\end{center}
\begin{enumerate}
	\item \textbf{Find the derivative:}
	\begin{enumerate}
		\item \(2x^5\)
		\item \(\ln(2x^5)\)
		\item \(e^{x-4}\)
		\item \(e^{x^2}\)
		\item \(\tan(x)\)
	\end{enumerate}
	\item \textbf{Find the derivative using the product rule:}
	\begin{enumerate}
		\item \(x\sin(x)\)
		\item \(\ln(x^{2x})\)
		\item \(xe^x\)
		\item \(\cos(x)(x+5)^2\)
	\end{enumerate}
	\item \textbf{Find the derivative using the quotient rule:}
	\begin{enumerate}
		\item \(\displaystyle\frac{x^2}{\sin(x)}\)
		\item \(\displaystyle\frac{e^5x}{x^2+2x-9}\)
		\item \(\cot(x)\)
		\item \(\displaystyle\frac{\ln(x)}{xe^{2x}}\)
	\end{enumerate}
	\item \textbf{Find the derivative using the chain rule:}
	\begin{enumerate}
		\item \(\cos(3x^2)\)
		\item \(\sin(\ln(x))\)
		\item \(\ln(\sec(x))\)
		\item \(\sin(x^2+2x)\)
	\end{enumerate}
\end{enumerate}
